import java.util.Scanner;

public class Calculator {
    public static void main(String[] args) {
        Scanner Enter = new Scanner(System.in);
        var a = 0;
        var b = 0;
        String Action1 = "";
        do {
            String input = Enter.nextLine();
            if (input.equals("")) {
                break;
            }
            String[] strings = input.trim().split("\\s+");
            if (strings.length != 3) {
                System.out.println("Ошибка" + "\n" + "Не соответствует условиям ввода");
                continue;
            }
            try {
                a = Integer.parseInt(strings[0]);
                b = Integer.parseInt(strings[2]);
                Action1 = strings[1];
                if (!Action1.matches("[+ * / -]")) {
                    System.out.println("Ошибка" + "\n" + "Не соответствуеЕЕЕт условиям ввода");
                    continue;
                }
                if (a > 10 || b > 10 || a < 0 || b < 0){
                    System.out.println("Ошибка" + "\n" + "Не соответствует условиям ввода");
                    continue;
                }
                System.out.println("Результат:" + "\n" + Action(a, b, Action1));
            } catch (Exception e) {
                System.out.println("Ошибка");
            }
        } while (true);
    }
    private static int Action(int a, int b, String Action1) {
        switch (Action1) {
            case "*":
                return a*b;
            case "+":
                return a+b;
            case "-":
                return a-b;
            case "/":
                return a/b;
            default:
                throw new IllegalArgumentException("Ошибка ввода");
        }
    }
}
